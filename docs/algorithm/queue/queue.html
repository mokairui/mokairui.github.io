# 数据结构 - 队列

模拟队列代码实现

public class Queue<T> implements Iterable<T> {
    // 记录首结点
    private Node head;
    // 记录最后一个结点
    private Node last;
    // 记录队列中元素的个数
    private int N;
    
    public Queue() {
        head = new Node(null, null);
        last = null;
        N = 0;
    }
    
    // 判断队列是否为空
    public boolean isEmpty(){
        return N == 0;
    }
    
    // 返回队列中元素的个数
    public int size(){
        return N;
    }
    
    // 向队列中插入元素t
    public void enqueue(T t){
        if (isEmpty()) {
            last = new Node(t, null);
            head.next = last;
        } else {
            // 实际用到记录尾节点last也就这个位置
            Node oldNode = this.last;
            last = new Node(t, null);
            oldNode.next = last;
        }
        
        N++;
    }

    // 从队列中拿出一个元素
    public T dequeue(){
        if (isEmpty()) {
            return null;
        }
        Node oldFirst = head.next;
        head.next = oldFirst.next;
        N--;
        if (isEmpty()) {
            last = null;
        }
        return oldFirst.item;
    }
    
    @Override
    public Iterator<T> iterator() {
        return new QIterator();
    }
    
    private class QIterator implements Iterator<T>{
        private Node n = head;
        @Override
        public boolean hasNext() {
            return n.next!=null;
        }
        @Override
        public T next() {
            Node node = n.next;
            n = n.next;
            return node.item;
        }
    }
    private class Node{
        public T item;
        public Node next;
        public Node(T item, Node next) {
            this.item = item;
            this.next = next;
        }
    }
}

测试

public class QueueTest {
    private static Queue<String> queue = new Queue<>();

    @Before
    public void init() {
        queue.enqueue("姚明");
        queue.enqueue("科比");
        queue.enqueue("麦迪");
        queue.enqueue("詹姆斯");

        for (String s : queue) {
            System.out.println(s);
        }
        System.out.println("----------------------------------");
    }

    @Test
    public void test() {
        System.out.println("出队列结果为:" + queue.dequeue());
        System.out.println("出队列结果为:" + queue.dequeue());
        System.out.println("出队列结果为:" + queue.dequeue());
        System.out.println("判断队列空间是否为空:" + queue.isEmpty());
        System.out.println("出队列结果为:" + queue.dequeue());
        System.out.println("队列中元素::" + queue.size());
    }
}
