# 数据结构 - 双向链表

模拟代码实现

public class BidirectionalLinkList<T> implements Iterable<T> {
    //首结点
    private Node head;
    //最后一个结点
    private Node last;
    //链表的长度
    private int N;
    
    public BidirectionalLinkList() {
        // 初始化头节点和尾节点
        this.head = new Node(null, null, null);
        this.last = null;
        // 初始化元素个数
        this.N = 0;
    }
    
    //清空链表
    public void clear(){
        this.head.next = null;
        this.last = null;
        this.N = 0;
    }
    
    //获取链表长度
    public int length(){
        return N;
    }
    
    //判断链表是否为空
    public boolean isEmpty(){
        return N==0;
    }
    
    //插入元素t
    public void insert(T t){
        if (isEmpty()) {
            // 如果链表为空
            // 创建新的节点
            Node newNode = new Node(t, head, null);
            // 让新节点称为尾节点
            last = newNode;
            // 让头节点指向尾节点
            head.next = last;
        } else {
            // 如果链表不是空
            Node oldLast = this.last;
            // 创建新的节点
            Node newNode = new Node(t, oldLast, null);
            // 让当前的尾节点指向新的节点
            oldLast.next = newNode;
            // 让新节点称为尾节点
            this.last = newNode;
        }
        // 元素个数+1
        this.N++;
    }
    
    //向指定位置i处插入元素t
    public void insert(int i,T t){
        // 找到i位置的前一个节点
        Node pre = head;
        for (int index = 0; index < i; index++) {
            pre = pre.next;
        }
        // 找到i位置的节点
        Node curr = pre.next;
        // 创建新的节点
        Node newNode = new Node(t, pre, curr);
        // 让i位置的前一个节点的下一个节点变为新的节点
        pre.next = newNode;
        // 让i位置的前一个节点变为新节点
        curr.pre = newNode;
        // 元素个数+1
        this.N++;
    }
    
    //获取指定位置i处的元素
    public T get(int i) {
        Node n = head.next;
        for (int index = 0; index < i; index++) {
            n = n.next;
        }
        return n.item;
    }
    
    //找到元素t在链表中第一次出现的位置
    public int indexOf(T t){
        Node node = this.head;
        for (int index = 0; node.next != null; index++) {
            node = node.next;
            if (node.item.equals(t)) {
                return index;
            }
        }
        return -1;
    }
    
    //删除位置i处的元素，并返回该元素
    public T remove(int i){
        // 找到i位置的前一个节点
        Node pre = this.head;
        for (int index = 0; index < i; index++) {
            pre = pre.next;
        }
        // 找到i位置的节点
        Node curr = pre.next;
        // 找打i位置的下一个节点
        Node next = curr.next;
        // 让i位置的前一个节点的下一个节点的位置变为i位置的下一个节点
        pre.next = next;
        // 让i位置的下一个节点的上一个节点的位置变为i位置的前一个节点
        next.pre = pre;
        return curr.item;
    }
    
    //获取第一个元素
    public T getFirst(){
        if (isEmpty()) {
            return null;
        }
        return this.head.next.item;
    }
    
    //获取最后一个元素
    public T getLast(){
        if (isEmpty()) {
            return null;
        }
        return last.item;
    }
    
    @Override
    public Iterator<T> iterator() {
        return new BiLinkList();
    }
    
    private class BiLinkList implements Iterator {
        private Node n;

        public BiLinkList() {
            this.n = head;
        }

        @Override
        public boolean hasNext() {
            return n.next != null;
        }
        
        @Override
        public Object next() {
            n = n.next;
            return n.item;
        }
    }
    
    //结点类
    private class Node{
        public Node(T item, Node pre, Node next) {
            this.item = item;
            this.pre = pre;
            this.next = next;
        }
        //存储数据
        public T item;
        //指向上一个结点
        public Node pre;
        //指向下一个结点
        public Node next;
    }
}

测试

public class BidirectionalLinkListTest {
    private static BidirectionalLinkList<String> directionalLinkList = new BidirectionalLinkList<>();
    
    @Before
    public void init() {
        directionalLinkList.insert("姚明");
        directionalLinkList.insert("科比");
        directionalLinkList.insert("麦迪");
        directionalLinkList.insert(1, "詹姆斯");

        for (String s : directionalLinkList) {
            System.out.println(s);
        }
        System.out.println("----------------------------------");
    }
    
    @Test
    public void test() {
        String getResult = directionalLinkList.get(1);
        System.out.println("获取索引1处的结果为:" + getResult);
        String removeResult = directionalLinkList.remove(0);
        System.out.println("删除索引0处的结果为:" + removeResult);
        System.out.println("第一个元素是:" + directionalLinkList.getFirst());
        System.out.println("最后一个元素是:" + directionalLinkList.getLast());
        directionalLinkList.clear();
        System.out.println("清空后的线性表中的元素个数为:" + directionalLinkList.length());
    }
}
